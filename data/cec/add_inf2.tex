\begin{itemize}
    \item[] A + -A = $\infty$
    \item[] A + $\infty$ = A
    \item[] $\infty$ + B = B
    \item[] Else
    \begin{itemize}
        \item[] $\lambda = \frac{B_y - A_y} {B_x - A_x}$
        \item[] $C_x = \lambda^2 - A_x - B_x$
        \item[] $C_y = \lambda(A_x - C_x) - A_y$
    \end{itemize}
\end{itemize}
